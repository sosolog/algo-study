from itertools import permutations
def sosu(n):
    if n < 2:
        return False
    
    for i in range(2, n//2+1):
        if n%i == 0:
            return False
    return True

def solution(numbers):
    answer=[]
    answer1=0
    for i in range(len(numbers)):
        answer.extend(permutations(numbers,i+1))
        result = [int(''.join(i)) for i in answer]
    
    for i in set(result):
        if sosu(i):
            answer1+=1
            
    return answer1
