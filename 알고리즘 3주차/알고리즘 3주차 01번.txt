from collections import deque
import sys
read = sys.stdin.readline
arr=[]

stack=[v]

    a, b = map(int, read().split())
    arr=[a,b]
for i of arr:
    if i[0] ==v:
        arr1=stack.pop(0)
        stack.push(i[1])
        v=i[1]

        stack.push(i[0])
